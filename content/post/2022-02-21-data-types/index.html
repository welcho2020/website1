Concatenation

Converting between types

x <- 1:3
x
## [1] 1 2 3
typeof(x)
## [1] "integer"
y <- as.character(x)
y
## [1] "1" "2" "3"
typeof(y)
## [1] "character"
x <- c(TRUE, FALSE)
x
## [1]  TRUE FALSE
typeof(x)
## [1] "logical"
y <- as.numeric(x)
y
## [1] 1 0
typeof(y)
## [1] "double"

mean(x)
## [1] NA
mean(x, na.rm = TRUE)
## [1] 2.5
summary(x)
